114  void Write(uint32_t address, std::span<const uint8_t> data) override
115  {
116  constexpr bit::Mask kLower6Bits = bit::MaskFromRange(0, 5);
117  constexpr bit::Mask kUpper6Bits = bit::MaskFromRange(6, 11);
118 
119  address = bit::Clear(address, kAddressMask);
120 
121  // Because the EEPROM uses 32-bit communication, write_data will be cast
122  // into a uint32_t *
123  std::span<const uint32_t> write_data(
124  reinterpret_cast<const uint32_t *>(data.data()),
125  data.size() / sizeof(uint32_t));
126  // const uint32_t * write_data = reinterpret_cast<const uint32_t *>(data);
127 
128  // The first 6 bits in the address (MSB) dictate which page is being written
129  // to in the EEPROM, and the last 6 bits (LSB) dictate offset in the page
130  uint32_t page_count = bit::Extract(address, kUpper6Bits);
131  uint32_t page_offset = bit::Extract(address, kLower6Bits);
132 
133  uint16_t eeprom_address = 0;
134 
135  for (const auto & word : write_data)
136  {
137  // Page offset is incremented by 4 because we're writing 32 bits
138  constexpr uint8_t kOffsetInterval = 4;
139 
140  eeprom_address = static_cast<uint16_t>((page_count << 6) + page_offset);
141 
142  eeprom_register->ADDR = eeprom_address;
143  eeprom_register->CMD = kWrite32Bits;
144  eeprom_register->WDATA = word;
145 
146  // Poll status register bit to see when writing is finished
147  auto check_register = []() {
148  return bit::Read(eeprom_register->INT_STATUS,
149  StatusRegister::kReadWriteStatusMask);
150  };
151 
152  auto timeout_status = Wait(kMaxTimeout, check_register);
153  if (!timeout_status)
154  {
155  throw Exception(std::errc::timed_out,
156  "Could not write to EEPROM in time.");
157  }
158 
159  // Clear write interrupt
160  eeprom_register->INT_CLR_STATUS =
161  (bit::Set(0, StatusRegister::kReadWriteStatusMask));
162 
163  page_offset += kOffsetInterval;
164 
165  // If the 64 byte page buffer fills up, then it must be
166  // programmed to the EEPROM before continuing.
167  if (page_offset == 64)
168  {
169  Program(eeprom_address);
170  page_count++;
171  page_offset = 0;
172  }
173  }
174 
175  // Program final information so that it's stored in the EEPROM
176  Program(eeprom_address);
177  }

179  void Read(uint32_t address, std::span<uint8_t> data) override
180  {
181  address = bit::Insert(address, 0b00, kAddressMask);

183  // Because the EEPROM uses 32-bit communication, read_data will be casted

186  data.size() / sizeof(uint32_t));
187 
188  int index = 0;
189  for (auto & word : read_data)
190  {
191  eeprom_register->ADDR = address + (index * 4);
192  eeprom_register->CMD = kRead32Bits;
193  word = eeprom_register->RDATA;
194  index++;
195  }
196  }
