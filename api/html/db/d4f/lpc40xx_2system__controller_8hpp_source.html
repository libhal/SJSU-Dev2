442  void Initialize() override
443  {
444  LPC_SC_TypeDef * sys = system_controller;
445  auto config = clock_configuration_;
446 
447  units::frequency::hertz_t system_clock = 0_Hz;
448  units::frequency::hertz_t pll0 = 0_Hz;
449  units::frequency::hertz_t pll1 = 0_Hz;
450 
451  // Reset all of the cached clock frequency values
452  units::frequency::hertz_t cpu = 0_Hz;
453  units::frequency::hertz_t usb = 0_Hz;
454  units::frequency::hertz_t spifi = 0_Hz;
455 
456  // =========================================================================
457  // Step 1. Select IRC as clock source for everything.
458  // Make sure PLLs are not clock sources for everything.
459  // =========================================================================
460  // Set CPU clock to system clock
461  CpuClockRegister::Register()
462  .Insert(Value(CpuClockSelect::kSystemClock), CpuClockRegister::kSelect)
463  .Save();
464  // Set USB clock to system clock
465  UsbClockRegister::Register()
466  .Insert(Value(UsbClockSelect::kSystemClock), UsbClockRegister::kSelect)
467  .Save();
468  // Set SPIFI clock to system clock
469  SpiFiClockRegister::Register()
470  .Insert(Value(SpifiClockSelect::kSystemClock),
471  SpiFiClockRegister::kSelect)
472  .Save();
473 
474  // Set the clock source to IRC and not external oscillator. The next phase
475  // disables that clock source, which will stop the system if this is not
476  // switched.
477  sys->CLKSRCSEL = Value(OscillatorSource::kIrc);
478 
479  // =========================================================================
480  // Step 2. Disable PLLs
481  // =========================================================================
482  // NOTE: The only bit in this register that is used is bit 0 which indicates
483  // enabled or disabled status, thus a single assignment is needed.
484  sys->PLL0CON = 0;
485  sys->PLL1CON = 0;
486  // Disabling external oscillator if it is not going to be used
487  OscillatorRegister::Register()
488  .Clear(OscillatorRegister::kExternalEnable)
489  .Save();
490 
491  // =========================================================================
492  // Step 3. Select oscillator source for System Clock and Main PLL
493  // =========================================================================
494  // Enable the external oscillator if we are using it, which would be the
495  // case if the alternative PLL is enabled or external oscillator is
496  // selected.
497  if (config.system_oscillator == OscillatorSource::kExternal ||
498  config.pll[1].enabled)
499  {
500  EnableExternalOscillator();
501  }
502 
503  sys->CLKSRCSEL = Value(config.system_oscillator);
504 
505  switch (config.system_oscillator)
506  {
507  case OscillatorSource::kIrc: system_clock = kDefaultIRCFrequency; break;
508  case OscillatorSource::kExternal:
509  system_clock = config.external_oscillator_frequency;
510  break;
511  }
512 
513  // =========================================================================
514  // Step 4. Configure PLLs
515  // =========================================================================
516  pll0 = SetupPll(&sys->PLL0CON, &sys->PLL0CFG, &sys->PLL0FEED,
517  &sys->PLL0STAT, 0);
518 
519  pll1 = SetupPll(&sys->PLL1CON, &sys->PLL1CFG, &sys->PLL1FEED,
520  &sys->PLL1STAT, 1);
521 
522  // =========================================================================
523  // Step 5. Set clock dividers for each clock source
524  // =========================================================================
525  // Set CPU clock divider
526  CpuClockRegister::Register()
527  .Insert(config.cpu.divider, CpuClockRegister::kDivider)
528  .Save();
529  // Set EMC clock divider
530  EmcClockRegister::Register()
531  .Insert(Value(config.emc_divider), EmcClockRegister::kDivider)
532  .Save();
533  // Set Peripheral clock divider
534  PeripheralClockRegister::Register()
535  .Insert(config.peripheral_divider, PeripheralClockRegister::kDivider)
536  .Save();
537  // Set USB clock divider
538  UsbClockRegister::Register()
539  .Insert(Value(config.usb.divider), UsbClockRegister::kDivider)
540  .Save();
541  // Set SPIFI clock divider
542  SpiFiClockRegister::Register()
543  .Insert(config.spifi.divider, SpiFiClockRegister::kDivider)
544  .Save();
545 
546  switch (config.cpu.clock)
547  {
548  case CpuClockSelect::kSystemClock: cpu = system_clock; break;
549  case CpuClockSelect::kPll0: cpu = pll0; break;
550  }
551 
552  switch (config.usb.clock)
553  {
554  case UsbClockSelect::kSystemClock: usb = system_clock; break;
555  case UsbClockSelect::kPll0: usb = pll0; break;
556  case UsbClockSelect::kPll1: usb = pll1; break;
557  }
558 
559  switch (config.spifi.clock)
560  {
561  case SpifiClockSelect::kSystemClock: spifi = system_clock; break;
562  case SpifiClockSelect::kPll0: spifi = pll0; break;
563  case SpifiClockSelect::kPll1: spifi = pll1; break;
564  }
565 
566  cpu_clock_rate_ = cpu / config.cpu.divider;
567  peripheral_clock_rate_ = cpu / config.peripheral_divider;
568  emc_clock_rate_ = cpu / (Value(config.emc_divider) + 1);
569  usb_clock_rate_ = usb / Value(config.usb.divider);
570  spifi_clock_rate_ = spifi / config.spifi.divider;
571 
572  // =========================================================================
573  // Step 6. Configure flash cycles per load
574  // =========================================================================
575  sys->PBOOST = 0b00;
576 
577  if (cpu_clock_rate_ < 20_MHz)
578  {
579  sys->FLASHCFG = Value(FlashConfiguration::kClock1);
580  }
581  else if (20_MHz <= cpu_clock_rate_ && cpu_clock_rate_ < 40_MHz)
582  {
583  sys->FLASHCFG = Value(FlashConfiguration::kClock2);
584  }
585  else if (40_MHz <= cpu_clock_rate_ && cpu_clock_rate_ < 60_MHz)
586  {
587  sys->FLASHCFG = Value(FlashConfiguration::kClock3);
588  }
589  else if (60_MHz <= cpu_clock_rate_ && cpu_clock_rate_ < 80_MHz)
590  {
591  sys->FLASHCFG = Value(FlashConfiguration::kClock4);
592  }
593  else if (80_MHz <= cpu_clock_rate_ && cpu_clock_rate_ < 100_MHz)
594  {
595  sys->FLASHCFG = Value(FlashConfiguration::kClock5);
596  }
597  else if (cpu_clock_rate_ >= 100_MHz)
598  {
599  sys->FLASHCFG = Value(FlashConfiguration::kClock5);
600  sys->PBOOST = 0b11;
601  }
602 
603  // =========================================================================
604  // Step 7. Finally select the sources for each clock
605  // =========================================================================
606  // Set CPU clock the source defined in the configuration
607  CpuClockRegister::Register()
608  .Insert(Value(config.cpu.clock), CpuClockRegister::kSelect)
609  .Save();
610 
611  // Set USB clock the source defined in the configuration
612  UsbClockRegister::Register()
613  .Insert(Value(config.usb.clock), UsbClockRegister::kSelect)
614  .Save();
615 
616  // Set SPIFI clock the source defined in the configuration
617  SpiFiClockRegister::Register()
618  .Insert(Value(config.spifi.clock), SpiFiClockRegister::kSelect)
619  .Save();
620  }

622  private:
623  units::frequency::hertz_t SetupPll(volatile uint32_t * control,
624  volatile uint32_t * config,
625  volatile uint32_t * feed,
626  volatile uint32_t * stat,
627  int pll_index) const
628  {
629  const auto & pll_config = clock_configuration_.pll[pll_index];
630  units::frequency::hertz_t fcco = 0_Hz;
631 
632  if (pll_config.enabled)
633  {
634  *config = bit::Insert(*config, pll_config.multiply - 1,
635  PllRegister::kMultiplier);
636 
637  if (clock_configuration_.system_oscillator == OscillatorSource::kIrc &&
638  pll_index == 0)
639  {
640  fcco = kDefaultIRCFrequency * pll_config.multiply;
641  }
642  else
643  {
644  fcco = clock_configuration_.external_oscillator_frequency *
645  pll_config.multiply;
646  }
647 
648  // In the datasheet this is the divider, but it acts to multiply the
649  // frequency higher to a point where the fcco is stable.
650  //
651  // fcco must be between 156 MHz to 320 MHz.
652  uint32_t fcco_divide = 0;
653  for (int divide_codes : { 0, 1, 2, 3 })
654  {
655  // Multiply the fcco by 2^divide_code
656  units::frequency::hertz_t final_fcco = fcco * (1 << divide_codes);
657  if (156_MHz <= final_fcco && final_fcco <= 320_MHz)
658  {
659  fcco_divide = divide_codes;
660  break;
661  }
662  }
663 
664  SJ2_ASSERT_FATAL(
665  fcco_divide != 0,
666  "The provided multiply value and oscillator choices results in "
667  "FCCO frequency outside of 156 MHz and 320 MHz. See page 65 "
668  "UM10562 LPC408x/407x User manual for more details.");
669 
670  *config = bit::Insert(*config, fcco_divide, PllRegister::kDivider);
671  // Enable PLL
672  *control = 1;
673  // Feed PLL in order to start the locking process
674  *feed = 0xAA;
675  *feed = 0x55;
676 
677  while (!bit::Read(*stat, PllRegister::kPllLockStatus))
678  {
679  continue;
680  }
681  }
682 
683  return fcco;
684  }

686  void EnableExternalOscillator() const
687  {
688  auto scs_register = OscillatorRegister::Register();
689  auto frequency = clock_configuration_.external_oscillator_frequency;
690  if (1_MHz <= frequency && frequency <= 20_MHz)
691  {
692  scs_register.Clear(OscillatorRegister::kRangeSelect);
693  }
694  else if (20_MHz <= frequency && frequency <= 25_MHz)
695  {
696  scs_register.Set(OscillatorRegister::kRangeSelect);
697  }
698  else
699  {
700  SJ2_ASSERT_FATAL(
701  false,
702  "External Oscillator Frequency is outside of the the acceptable 1 "
703  "MHz <--> 25 MHz.");
704  }
705 
706  scs_register.Set(OscillatorRegister::kExternalEnable).Save();
707 
708  while (!scs_register.Read(OscillatorRegister::kExternalReady))
709  {
710  continue;
711  }
712  }
713 
714  ClockConfiguration & clock_configuration_;
715  units::frequency::hertz_t cpu_clock_rate_ = 0_Hz;
716  units::frequency::hertz_t peripheral_clock_rate_ = 0_Hz;
717  units::frequency::hertz_t emc_clock_rate_ = 0_Hz;
718  units::frequency::hertz_t usb_clock_rate_ = 0_Hz;
719  units::frequency::hertz_t spifi_clock_rate_ = 0_Hz;
