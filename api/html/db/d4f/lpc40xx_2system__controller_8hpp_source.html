241  uint32_t CalculatePll(units::frequency::megahertz_t input_frequency,
242  units::frequency::megahertz_t desired_frequency) const
243  {
244  SJ2_ASSERT_FATAL(desired_frequency < 384_MHz && desired_frequency > 12_MHz,
245  "Frequency must be lower than 384 MHz"
246  "and greater than or equal to 12 MHz");
247  bool calculating = true;
248  uint32_t multiplier_value;
249  if ((desired_frequency.to<uint32_t>() % input_frequency.to<uint32_t>()) > 0)
250  {
251  multiplier_value =
252  static_cast<uint32_t>((desired_frequency / input_frequency) + 1);
253  }
254  else
255  {
256  multiplier_value =
257  static_cast<uint32_t>(desired_frequency / input_frequency);
258  }
259  uint16_t divider_value = 1;
260  while (calculating)
261  {
262  uint16_t current_controlled_oscillator_frequency;
263  current_controlled_oscillator_frequency = static_cast<uint16_t>(
264  (static_cast<uint16_t>(input_frequency) * multiplier_value * 2) *
265  divider_value);
266  if (current_controlled_oscillator_frequency >= 156)
267  {
268  calculating = false;
269  }
270  else
271  {
272  divider_value = static_cast<uint16_t>(divider_value * 2);
273  SJ2_ASSERT_FATAL(divider_value < 8,
274  "PLL divider value went out of bounds");
275  }
276  }
277 
278  return multiplier_value;
279  }

281  void SetMainPll(units::frequency::megahertz_t input_frequency,
282  units::frequency::megahertz_t desired_frequency) const
283  {
284  uint16_t divider_value = 1;
285  uint32_t multiplier_value =
286  CalculatePll(input_frequency, desired_frequency);
287  // units::frequency::megahertz_t actual_speed =
288  // input_frequency * multiplier_value;
289  // TO DO: use registers to retreive values
290  SelectOscillatorSource(OscillatorSource::kIrc);
291  SelectMainClockSource(MainClockSource::kBaseClock);
292  SelectUsbClockSource(UsbSource::kBaseClock);
293  SelectSpifiClockSource(SpifiSource::kBaseClock);
294  // must subtract 1 from multiplier value as specified in datasheet
295  system_controller->PLL0CFG =
296  (system_controller->PLL0CFG & ~kClearPllMultiplier) |
297  (multiplier_value - 1);
298  system_controller->PLL0CFG =
299  (system_controller->PLL0CFG & ~kClearPllDivider) | (divider_value << 5);
300  system_controller->PLL0CON |= kEnablePll;
301  // Necessary feed sequence to ensure the changes are intentional
302  system_controller->PLL0FEED = 0xAA;
303  system_controller->PLL0FEED = 0x55;
304 
305  while (!bit::Read(system_controller->PLL0STAT, kPlock))
306  {
307  continue;
308  }
309  }
