117  Status GetDistance(units::length::millimeter_t * distance) override
118  {
119  std::array<uint8_t, kDeviceDataLength> device_data = { 0 };

121  uart_pin_.Write(kPromptMeasurementCommand, kCommandLength);

123 
124  if (device_data[0] != kFrameHeader || device_data[1] != kFrameHeader)
125  {
