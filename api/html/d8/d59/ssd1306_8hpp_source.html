143  void DrawPixel(int32_t x, int32_t y, Color_t color) override
144  {
145  // The 3 least significant bits hold the bit position within the byte
146  uint32_t bit_position = y & 0b111;
147 
148  // Each byte makes up a vertical column.
149  // Shifting by 3, which also divides by 8 (the 8-bits of a column), will
150  // be the row that we need to edit.
151  uint32_t row = y >> 3;
152 
153  // Mask to clear the bit
154  uint32_t clear_mask = ~(1 << bit_position);
155 
156  // Mask to set the bit, if color.alpha != 0
157  bool pixel_is_on = !color.IsBlank();
158  uint32_t set_mask = pixel_is_on << bit_position;
159 
160  // Address of the pixel column to edit
161  uint8_t * pixel_column = &(bitmap_[row][x]);
162 
163  // Read pixel column and update the pixel
164  uint32_t result = (*pixel_column & clear_mask) | set_mask;
165 
166  // Update pixel with the result of this operation
167  *pixel_column = static_cast<uint8_t>(result);
168  }

279  void Write(uint32_t data, Transaction transaction, size_t size = 1)
280  {
281  dc_.Set(static_cast<sjsu::Gpio::State>(transaction));
282  cs_.Set(sjsu::Gpio::State::kLow);
283 
284  bit::Mask mask = {
285  .position = static_cast<uint32_t>((size - 1) * 8),
286  .width = 8,
287  };
288 
289  for (size_t i = 0; i < size; i++)
290  {
291  spi_.Transfer(static_cast<uint8_t>(bit::Extract(data, mask)));
292  mask = mask >> 8;
293  }
294 
295  cs_.Set(sjsu::Gpio::State::kHigh);
296  }
