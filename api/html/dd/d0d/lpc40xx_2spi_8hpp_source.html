150  Status Initialize() const override
151  {
152  constexpr uint8_t kSpiFormatCode = 0b00;
153 
154  // Power up peripheral
155  system_controller_.PowerUpPeripheral(bus_.power_on_bit);
156  // Enable SSP pins
157  bus_.mosi.SetPinFunction(bus_.pin_function_id);
158  bus_.miso.SetPinFunction(bus_.pin_function_id);
159  bus_.sck.SetPinFunction(bus_.pin_function_id);
160  // Set SSP frame format to SPI
161  bus_.registers->CR0 = bit::Insert(
162  bus_.registers->CR0, kSpiFormatCode, ControlRegister0::kFrameBit);
163  // Set SPI to master mode by clearing
164  bus_.registers->CR1 = bit::Clear(bus_.registers->CR1,
165  ControlRegister1::kSlaveModeBit.position);
166  // Enable SSP
167  bus_.registers->CR1 =
168  bit::Set(bus_.registers->CR1, ControlRegister1::kSpiEnable.position);
169  return Status::kSuccess;
170  }

230  void SetClock(units::frequency::hertz_t frequency,
231  bool positive_clock_on_idle = false,
232  bool read_miso_on_rising = false) const override
233  {
234  bus_.registers->CR0 = bit::Insert(bus_.registers->CR0,
235  positive_clock_on_idle,
236  ControlRegister0::kPolarityBit);
237 
238  bus_.registers->CR0 = bit::Insert(
239  bus_.registers->CR0, read_miso_on_rising, ControlRegister0::kPhaseBit);
240 
241  uint16_t prescaler =
242  (system_controller_.GetPeripheralFrequency(bus_.power_on_bit) /
243  frequency)
244  .to<uint16_t>();
245  // Store lower half of precalar in clock prescalar register
246  bus_.registers->CPSR = prescaler & 0xFF;
247  // Store upper 8 bit half of the prescalar in control register 0
248  bus_.registers->CR0 = bit::Insert(
249  bus_.registers->CR0, prescaler >> 8, ControlRegister0::kDividerBit);
250  }
