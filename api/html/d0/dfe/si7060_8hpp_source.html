50  Status GetTemperature(units::temperature::celsius_t * temperature) override
51  {
52  // Note that: 1 << 14 = 2^14 = 16384
53  constexpr int32_t kSubtractTemperatureData = 1 << 14;
54 
55  uint8_t most_significant_register;
56  uint8_t least_significant_register;
57  // The register will enable the device to collect data once
58  // and automatically sets the stop bit to 0 (2nd bit).
59  i2c_.Write(address_, { kOneBurstRegister, 0x04 });
60  // Auto increments I2c register address pointer.
61  i2c_.Write(address_, { kAutomaticBitRegister, 0x01 });
62  i2c_.WriteThenRead(
63  address_, { kMostSignificantRegister }, &most_significant_register, 1);
64  i2c_.Read(address_, &least_significant_register, 1);
65 
66  // The write and read operation sets the most significant bit to one,
67  // telling the register that the data has been read.
68  // Therefore must clear the most significant bit from above.
69  most_significant_register = bit::Clear(most_significant_register, 7);
70  // Combine temperature register data
71  int32_t temperature_data =
72  (most_significant_register << 8) | least_significant_register;
73  // The required computation after bit shifting.
74  // Formula can be found below, see page 4:
75  // datasheets/Temperature-Sensor/si7060-datasheets.pdf
76  float acquired_temperature =
77  static_cast<float>(temperature_data - kSubtractTemperatureData);
78  *temperature =
79  units::temperature::celsius_t((acquired_temperature / 160.0f) + 55.0f);
80 
81  return Status::kSuccess;
82  }
