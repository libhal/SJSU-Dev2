398 inline void PollingVerification(
399  PollVerificationParameter_t poll,
400  const std::experimental::source_location & location =
401  std::experimental::source_location::current())
402 {
403  // Setup
404  // Setup: Create a polling variable that will track if the polling function is
405  // still polling or not. This is set to false after the polling
406  // function is finished. This variable only become false the polling
407  // function was never locked in a loop or if environmental change due
408  // to the release function results in the polling_function finishing.
409  bool polling = true;
410 
411  // Setup: First run the function that will setup the environment to cause the
412  // polling function to poll/loop/lock.
413  poll.locking_function();
414 
415  std::string file_location =
416  location.file_name() + std::string(":") + std::to_string(location.line());
417 
418  std::string lock_failure_string = file_location;
419  lock_failure_string +=
420  ": locking_function() did not result in the polling_function() locking. "
421  "Failing this test!";
422 
423  std::string release_failure_string = file_location;
424  release_failure_string +=
425  ": polling_function() was not released by release_function(). Failing "
426  "this test!";
427 
428  // Exercise
429  // Exercise: Create the additional thread that will release the
430  // polling_function() from polling after delay_time has elapsed.
431  // Will also verify that that polling variable is correct.
432  std::thread polling_release_thread(
433  [&poll, &polling, &lock_failure_string, &release_failure_string]() {
434  // Wait for polling function to begin polling
435  std::this_thread::sleep_for(poll.delay_time);
436 
437  // Check that polling variable is still TRUE (will become false if
438  // polling finished early).
439  REQUIRE_MESSAGE(polling, lock_failure_string.c_str());
440 
441  // Attempt to release the polling function from polling.
442  poll.release_function();
443 
444  // Wait for polling function to finish.
445  std::this_thread::sleep_for(poll.delay_time);
446 
447  // Check that polling variable is now FALSE
448  REQUIRE_MESSAGE(!polling, release_failure_string.c_str());
449  });
450 
451  // Exercise: Begin polling
452  poll.polling_function();
453 
454  // Polling function has finished at this point, set polling to false.
455  polling = false;
456 
457  // Wait for polling thread to finish.
458  polling_release_thread.join();
459 }
