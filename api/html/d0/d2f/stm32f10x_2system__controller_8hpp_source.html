338  void Initialize() override
339  {
340  units::frequency::hertz_t system_clock = 0_Hz;
341 
342  // =========================================================================
343  // Step 1. Select internal clock source for everything.
344  // Make sure PLLs are not clock sources for everything.
345  // =========================================================================
346  // Step 1.1 Set SystemClock to HSI
347  clock_control->CFGR = bit::Insert(
348  clock_control->CFGR, Value(SystemClockSelect::kHighSpeedInternal),
349  ClockConfigurationRegisters::kSystemClockSelect);
350 
351  // Step 1.4 Reset RTC clock registers
352  clock_control->BDCR =
353  bit::Set(clock_control->BDCR, RtcRegisters::kBackupDomainReset);
354 
355  // Manually clear the RTC reset bit
356  clock_control->BDCR =
357  bit::Clear(clock_control->BDCR, RtcRegisters::kBackupDomainReset);
358 
359  // =========================================================================
360  // Step 2. Disable PLL and external clock sources
361  // =========================================================================
362  // Step 2.1 Disable PLLs
363  clock_control->CR =
364  bit::Clear(clock_control->CR, ClockControlRegisters::kPllEnable);
365 
366  // Step 2.1 Disable External Oscillators
367  clock_control->CR = bit::Clear(clock_control->CR,
368  ClockControlRegisters::kExternalOscEnable);
369 
370  // =========================================================================
371  // Step 3. Enable External Oscillators
372  // =========================================================================
373  // Step 3.1 Enable High speed external Oscillator
374  if (config_.high_speed_external != 0_MHz)
375  {
376  clock_control->CR = bit::Set(clock_control->CR,
377  ClockControlRegisters::kExternalOscEnable);
378 
379  while (!bit::Read(clock_control->CR,
380  ClockControlRegisters::kExternalOscReady))
381  {
382  continue;
383  }
384  }
385 
386  // Step 3.2 Enable Low speed external Oscillator
387  if (config_.low_speed_external != 0_MHz)
388  {
389  clock_control->BDCR =
390  bit::Insert(clock_control->BDCR, 1, RtcRegisters::kLowSpeedOscEnable);
391 
392  while (!bit::Read(clock_control->BDCR, RtcRegisters::kLowSpeedOscReady))
393  {
394  continue;
395  }
396  }
397 
398  // =========================================================================
399  // Step 4. Set oscillator source for PLLs
400  // =========================================================================
401  if (config_.pll.source == PllSource::kHighSpeedExternalDividedBy2)
402  {
403  clock_control->CFGR = bit::Set(
404  clock_control->CFGR, ClockConfigurationRegisters::kHsePreDivider);
405  }
406  else
407  {
408  clock_control->CFGR = bit::Clear(
409  clock_control->CFGR, ClockConfigurationRegisters::kHsePreDivider);
410  }
411 
412  clock_control->CFGR =
413  bit::Insert(clock_control->CFGR, Value(config_.pll.source),
414  ClockConfigurationRegisters::kPllSource);
415 
416  // =========================================================================
417  // Step 5. Setup PLLs and enable them where necessary
418  // =========================================================================
419  if (config_.pll.enable)
420  {
421  clock_control->CFGR =
422  bit::Insert(clock_control->CFGR, Value(config_.pll.multiply),
423  ClockConfigurationRegisters::kPllMul);
424 
425  clock_control->CR =
426  bit::Set(clock_control->CR, ClockControlRegisters::kPllEnable);
427 
428  while (!bit::Read(clock_control->CR, ClockControlRegisters::kPllReady))
429  {
430  continue;
431  }
432 
433  switch (config_.pll.source)
434  {
435  case PllSource::kHighSpeedInternal:
436  pll_clock_rate_ = kHighSpeedInternal / 2;
437  break;
438  case PllSource::kHighSpeedExternal:
439  pll_clock_rate_ = config_.high_speed_external;
440  break;
441  case PllSource::kHighSpeedExternalDividedBy2:
442  pll_clock_rate_ = config_.high_speed_external / 2;
443  break;
444  }
445 
446  // Multiply the PLL clock up to the correct rate.
447  pll_clock_rate_ = pll_clock_rate_ * (Value(config_.pll.multiply) + 2);
448  }
449 
450  // =========================================================================
451  // Step 6. Setup peripheral dividers
452  // =========================================================================
453  // Step 6.1 Set USB divider
454  clock_control->CFGR =
455  bit::Insert(clock_control->CFGR, Value(config_.pll.usb.divider),
456  ClockConfigurationRegisters::kUsbPrescalar);
457 
458  // Step 6.2 Set AHB divider
459  clock_control->CFGR =
460  bit::Insert(clock_control->CFGR, Value(config_.ahb.divider),
461  ClockConfigurationRegisters::kAHBDivider);
462 
463  // Step 6.3 Set APB1 divider
464  clock_control->CFGR =
465  bit::Insert(clock_control->CFGR, Value(config_.ahb.apb1.divider),
466  ClockConfigurationRegisters::kAPB1Divider);
467 
468  // Step 6.4 Set APB2 divider
469  clock_control->CFGR =
470  bit::Insert(clock_control->CFGR, Value(config_.ahb.apb2.divider),
471  ClockConfigurationRegisters::kAPB2Divider);
472 
473  // Step 6.5 Set ADC divider
474  clock_control->CFGR =
475  bit::Insert(clock_control->CFGR, Value(config_.ahb.apb2.adc.divider),
476  ClockConfigurationRegisters::kAdcDivider);
477 
478  // =========================================================================
479  // Step 7. Set System Clock and RTC Clock
480  // =========================================================================
481  uint32_t target_clock_source = Value(config_.system_clock);
482 
483  // Step 7.1 Set the Flash wait states appropriately prior to setting the
484  // system clock frequency. Failure to do this will cause the system
485  // to be unable to read from flash, resulting in the platform
486  // locking up. See p.60 of RM0008 for the Flash ACR register
487  if (config_.system_clock == SystemClockSelect::kPll)
488  {
489  if (pll_clock_rate_ <= 24_MHz)
490  {
491  // 0 Wait states
492  flash->ACR = sjsu::bit::Insert(flash->ACR, 0b000,
493  sjsu::bit::CreateMaskFromRange(0, 2));
494  }
495  else if (24_MHz <= pll_clock_rate_ && pll_clock_rate_ <= 48_MHz)
496  {
497  // 1 Wait state
498  flash->ACR = sjsu::bit::Insert(flash->ACR, 0b001,
499  sjsu::bit::CreateMaskFromRange(0, 2));
500  }
501  else
502  {
503  // 2 Wait states
504  flash->ACR = sjsu::bit::Insert(flash->ACR, 0b010,
505  sjsu::bit::CreateMaskFromRange(0, 2));
506  }
507  }
508 
509  // Step 7.2 Set system clock source
510  // NOTE: return error if clock = SystemClockSelect::kHighSpeedExternal and
511  // high speed external is not enabled.
512  clock_control->CFGR =
513  bit::Insert(clock_control->CFGR, Value(config_.system_clock),
514  ClockConfigurationRegisters::kSystemClockSelect);
515 
516  while (bit::Extract(clock_control->CFGR,
517  ClockConfigurationRegisters::kSystemClockStatus) !=
518  target_clock_source)
519  {
520  continue;
521  }
522 
523  switch (config_.system_clock)
524  {
525  case SystemClockSelect::kHighSpeedInternal:
526  system_clock = kHighSpeedInternal;
527  break;
528  case SystemClockSelect::kHighSpeedExternal:
529  system_clock = config_.high_speed_external;
530  break;
531  case SystemClockSelect::kPll: system_clock = pll_clock_rate_; break;
532  }
533 
534  // Step 7.3 Set the RTC oscillator source
535  clock_control->BDCR =
536  bit::Insert(clock_control->BDCR, Value(config_.rtc.source),
537  RtcRegisters::kRtcSourceSelect);
538 
539  // Step 7.4 Enable/Disable the RTC
540  clock_control->BDCR = bit::Insert(clock_control->BDCR, config_.rtc.enable,
541  RtcRegisters::kRtcEnable);
542 
543  // =========================================================================
544  // Step 8. Define the clock rates for the system
545  // =========================================================================
546  switch (config_.ahb.divider)
547  {
548  case AHBDivider::kDivideBy1: ahb_clock_rate_ = system_clock / 1; break;
549  case AHBDivider::kDivideBy2: ahb_clock_rate_ = system_clock / 2; break;
550  case AHBDivider::kDivideBy4: ahb_clock_rate_ = system_clock / 4; break;
551  case AHBDivider::kDivideBy8: ahb_clock_rate_ = system_clock / 8; break;
552  case AHBDivider::kDivideBy16: ahb_clock_rate_ = system_clock / 16; break;
553  case AHBDivider::kDivideBy64: ahb_clock_rate_ = system_clock / 64; break;
554  case AHBDivider::kDivideBy128:
555  ahb_clock_rate_ = system_clock / 128;
556  break;
557  case AHBDivider::kDivideBy256:
558  ahb_clock_rate_ = system_clock / 256;
559  break;
560  case AHBDivider::kDivideBy512:
561  ahb_clock_rate_ = system_clock / 512;
562  break;
563  }
564 
565  switch (config_.ahb.apb1.divider)
566  {
567  case APBDivider::kDivideBy1:
568  apb1_clock_rate_ = ahb_clock_rate_ / 1;
569  break;
570  case APBDivider::kDivideBy2:
571  apb1_clock_rate_ = ahb_clock_rate_ / 2;
572  break;
573  case APBDivider::kDivideBy4:
574  apb1_clock_rate_ = ahb_clock_rate_ / 4;
575  break;
576  case APBDivider::kDivideBy8:
577  apb1_clock_rate_ = ahb_clock_rate_ / 8;
578  break;
579  case APBDivider::kDivideBy16:
580  apb1_clock_rate_ = ahb_clock_rate_ / 16;
581  break;
582  }
583 
584  switch (config_.ahb.apb2.divider)
585  {
586  case APBDivider::kDivideBy1:
587  apb2_clock_rate_ = ahb_clock_rate_ / 1;
588  break;
589  case APBDivider::kDivideBy2:
590  apb2_clock_rate_ = ahb_clock_rate_ / 2;
591  break;
592  case APBDivider::kDivideBy4:
593  apb2_clock_rate_ = ahb_clock_rate_ / 4;
594  break;
595  case APBDivider::kDivideBy8:
596  apb2_clock_rate_ = ahb_clock_rate_ / 8;
597  break;
598  case APBDivider::kDivideBy16:
599  apb2_clock_rate_ = ahb_clock_rate_ / 16;
600  break;
601  }
602 
603  switch (config_.rtc.source)
604  {
605  case RtcSource::kNoClock: rtc_clock_rate_ = 0_Hz; break;
606  case RtcSource::kLowSpeedInternal:
607  rtc_clock_rate_ = kLowSpeedInternal;
608  break;
609  case RtcSource::kLowSpeedExternal:
610  rtc_clock_rate_ = config_.low_speed_external;
611  break;
612  case RtcSource::kHighSpeedExternalDividedBy128:
613  rtc_clock_rate_ = config_.high_speed_external / 128;
614  break;
615  }
616 
617  switch (config_.pll.usb.divider)
618  {
619  case UsbDivider::kDivideBy1: usb_clock_rate_ = pll_clock_rate_; break;
620  case UsbDivider::kDivideBy1Point5:
621  usb_clock_rate_ = (pll_clock_rate_ * 2) / 3;
622  break;
623  }
624 
625  switch (config_.ahb.apb1.divider)
626  {
627  case APBDivider::kDivideBy1:
628  timer_apb1_clock_rate_ = apb1_clock_rate_;
629  break;
630  default: timer_apb1_clock_rate_ = apb1_clock_rate_ * 2; break;
631  }
632 
633  switch (config_.ahb.apb2.divider)
634  {
635  case APBDivider::kDivideBy1:
636  timer_apb2_clock_rate_ = apb2_clock_rate_;
637  break;
638  default: timer_apb2_clock_rate_ = apb2_clock_rate_ * 2; break;
639  }
640 
641  switch (config_.ahb.apb2.adc.divider)
642  {
643  case AdcDivider::kDivideBy2:
644  adc_clock_rate_ = apb2_clock_rate_ / 2;
645  break;
646  case AdcDivider::kDivideBy4:
647  adc_clock_rate_ = apb2_clock_rate_ / 4;
648  break;
649  case AdcDivider::kDivideBy6:
650  adc_clock_rate_ = apb2_clock_rate_ / 6;
651  break;
652  case AdcDivider::kDivideBy8:
653  adc_clock_rate_ = apb2_clock_rate_ / 8;
654  break;
655  }

776 inline void SetMaximumClockSpeed()
777 {
778  using sjsu::stm32f10x::SystemController;
779  auto & config =
780  sjsu::SystemController::GetPlatformController()
781  .GetClockConfiguration<SystemController::ClockConfiguration>();
782 
783  // Set the speed of the high speed external oscillator.
784  // NOTE: Change this if its different for your board.
785  config.high_speed_external = 8_MHz;
786  // Set the source of the PLL's oscillator to 4MHz.
787  // See page 93 in RM0008 to see that the internal high speed oscillator,
788  // which is 8MHz, is divided by 2 before being fed to the PLL to get 4MHz.
789  config.pll.source = SystemController::PllSource::kHighSpeedExternal;
790  // Enable PLL to increase the frequency of the system
791  config.pll.enable = true;
792  // Multiply the 8MHz * 9 => 72 MHz
793  config.pll.multiply = SystemController::PllMultiply::kMultiplyBy9;
794  // Set the system clock to the PLL
795  config.system_clock = SystemController::SystemClockSelect::kPll;
796  // APB1's maximum frequency is 36 MHz, so we divide 72 MHz / 2 => 36 MHz.
797  config.ahb.apb1.divider = SystemController::APBDivider::kDivideBy2;
798  // Keep APB1's clock undivided as it can handle up to 72 MHz.
799  config.ahb.apb2.divider = SystemController::APBDivider::kDivideBy1;
800  // Maximum frequency for ADC is 14 MHz, thus the best we can do is divide
801  // 72 MHz / 6 to get 12 MHz.
802  config.ahb.apb2.adc.divider = SystemController::AdcDivider::kDivideBy6;
803  // Set USB to divide by 1.5 which will give you 48 MHz which is needed for USB
804  // usage.
805  config.pll.usb.divider = SystemController::UsbDivider::kDivideBy1Point5;
806 
807  // Initialize platform with new clock configuration settings.
808  sjsu::InitializePlatform();
809 }
