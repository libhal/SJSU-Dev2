203  void ModuleInitialize() override
204  {
205  auto & system = sjsu::SystemController::GetPlatformController();
206 
207  // Step 0. Check if the peripheral is already powered on. If it is, then
208  // this peripheral has already been initialized and should not be
209  // initialized again.
210  if (system.IsPeripheralPoweredUp(SystemController::Peripherals::kAdc1))
211  {
212  return;
213  }
214 
215  // Step 1. Power on Peripherals
216  system.PowerUpPeripheral(SystemController::Peripherals::kAdc1);
217  system.PowerUpPeripheral(SystemController::Peripherals::kDma1);
218 
219  // Step 2. Set Power on ADC bit
220  bit::Register(&adc1->CR2).Set(Control2::kAdcOn).Save();
221  sjsu::Delay(1ms);
222 
223  // Step 3. Run Calibration on each peripheral
224  CalibrateADC();
225 
226  // Step 4. Setup DMA for collecting ADC samples
227  SetupDMA();
228 
229  // Step 5. Set ADC control registers and sequence registers
230  adc1->SQR3 = RegularSequence::kRegisterSequence3;
231  adc1->SQR2 = RegularSequence::kRegisterSequence2;
232  adc1->SQR1 = RegularSequence::kRegisterSequence1;
233  adc1->CR1 = Control1::kDefault;
234  adc1->CR2 = Control2::kDefault;
235 
236  // Step 6. Turn on ADC
237  bit::Register(&adc1->CR2).Set(Control2::kAdcOn).Save();
238 
239  // Step 7. Start the conversion process
240  bit::Register(&adc1->CR2).Set(Control2::kSoftwareStart).Save();
241 
242  // Give some time for samples to be collected before leaving initialize
243  Delay(1ms);
