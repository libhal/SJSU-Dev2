171  static void TimerHandler(const CaptureChannelPartial_t & channel)
172  {
173  constexpr bit::Mask kCaptureInterruptFlagBit = bit::MaskFromRange(4, 5);
174  constexpr uint8_t kResetCaptureInterrupts = 0b11;
175 
176  if (*channel.user_callback != nullptr)
177  {
178  CaptureStatus_t status;
179  status.flags = channel.timer_register->IR;
180 
181  if (*channel.channel_number == CaptureChannelNumber::kChannel0)
182  {
183  status.count = channel.timer_register->CR0;
184  }
185  else
186  {
187  status.count = channel.timer_register->CR1;
188  }
189 
190  (*channel.user_callback)(status);
191  }
192  channel.timer_register->IR = bit::Insert(channel.timer_register->IR,
193  kResetCaptureInterrupts,
194  kCaptureInterruptFlagBit);
195  }

214  void Initialize(CaptureCallback callback = nullptr,
215  int32_t interrupt_priority = -1) const override
216  {
217  if (frequency_ == 0_Hz)
218  {
219  throw Exception(
220  std::errc::invalid_argument,
221  "Cannot have zero ticks per microsecond, please choose 1 or more.");
222  }
223 
224  auto & system = sjsu::SystemController::GetPlatformController();
225  system.PowerUpPeripheral(timer_.channel.id);
226 
227  // Configure prescaler
228  uint32_t prescaler = system.GetClockRate(timer_.channel.id) / frequency_;
229  timer_.channel.timer_register->PR = prescaler;
230 
231  // Enable timer
232  constexpr bit::Mask kTimerEnableBit = bit::MaskFromRange(0);
233  timer_.channel.timer_register->TCR =
234  bit::Set(timer_.channel.timer_register->TCR, kTimerEnableBit.position);
235 
236  // Disable capture channel
237  ConfigureCapture(CaptureEdgeMode::kNone);
238  EnableCaptureInterrupt(false);
239 
240  // Install capture ISR
241  *timer_.channel.user_callback = callback;
242  sjsu::InterruptController::GetPlatformController().Enable(
243  { .interrupt_request_number = timer_.channel.irq,
244  .interrupt_handler = timer_.handler,
245  .priority = interrupt_priority });
246  }
