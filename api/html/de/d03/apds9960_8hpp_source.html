313  virtual void ProcessGestureData(uint8_t level)
314  {
315  constexpr uint8_t kMaxPointsIndex = 7;
316  // Read Gesture FIFOs (U/D/L/R)
317  // Store raw gesture data in a 4x32 uint8_t array
318  for (int i = 0; i < (4 * level); i += 4)
319  {
320  if (index_ > kMaxPointsIndex)
321  {
322  index_ = 0;
323  }
324  // Check which photodiode pair sensed more IR light
325  if (abs((gfifo_data_[i] - gfifo_data_[i + 1])) >
326  abs((gfifo_data_[i + 2] - gfifo_data_[i + 3])))
327  {
328  // If a photodiode has a smaller value,
329  // then that photodiode was covered and movement detected
330  if ((gfifo_data_[i] - gfifo_data_[i + 1]) < up_sensitivity_) // Up
331  {
332  points_[index_] = Direction::kUp;
333  index_++;
334  }
335  else if ((gfifo_data_[i] - gfifo_data_[i + 1]) >
336  down_sensitivity_) // Down
337  {
338  points_[index_] = Direction::kDown;
339  index_++;
340  }
341  }
342  else if (abs((gfifo_data_[i] - gfifo_data_[i + 1])) <
343  abs((gfifo_data_[i + 2] - gfifo_data_[i + 3])))
344  {
345  // If a photodiode has a smaller value,
346  // then that photodiode was covered and movement detected
347  if ((gfifo_data_[i + 2] - gfifo_data_[i + 3]) <
348  left_sensitivity_) // Left
349  {
350  points_[index_] = Direction::kLeft;
351  index_++;
352  }
353  else if ((gfifo_data_[i + 2] - gfifo_data_[i + 3]) >
354  right_sensitivity_) // Right
355  {
356  points_[index_] = Direction::kRight;
357  index_++;
358  }
359  }
360  // Check the sum of all photodiodes
361  if ((gfifo_data_[i] + gfifo_data_[i + 1] + gfifo_data_[i + 2] +
362  gfifo_data_[i + 3]) > near_sensitivity_) // Near
363  {
364  near_count_++;
365  }
366  else if ((gfifo_data_[i] + gfifo_data_[i + 1] + gfifo_data_[i + 2] +
367  gfifo_data_[i + 3]) < far_sensitivity_) // Far
368  {
369  far_count_++;
370  }
371  }

373  virtual Gesture DecodeGestureData(uint8_t level)
374  {
375  Gesture result = kError;
376  constexpr uint8_t kFarNearSensitivity = 15;
377  constexpr uint8_t kNearFarCountMinimum = 5;
378 
379  if (level == kMaxFIFOLevel)
380  {
381  result = kError;
382  }
383  else if (level > kFarNearSensitivity)
384  {
385  if (far_count_ > kNearFarCountMinimum)
386  {
387  result = kFAR;
388  }
389  else if (near_count_ > kNearFarCountMinimum)
390  {
391  result = kNEAR;
392  }
393  }
394  sjsu::Delay(30ms);
395  for (int i = 0; i < kMaxDataSize - 1; i++)
396  {
397  // Check initial point, then check next point
398  // to figure out the direction of the swipe
399  switch (points_[i])
400  {
401  case Direction::kUp:
402  if (points_[i + 1] == Direction::kDown)
403  {
404  result = kSwipeDOWN;
405  }
406  break;
407  case Direction::kDown:
408  if (points_[i + 1] == Direction::kUp)
409  {
410  result = kSwipeUP;
411  }
412  break;
413  case Direction::kLeft:
414  if (points_[i + 1] == Direction::kRight)
415  {
416  result = kSwipeRIGHT;
417  }
418  break;
419  case Direction::kRight:
420  if (points_[i + 1] == Direction::kLeft)
421  {
422  result = kSwipeLEFT;
423  }
424  break;
425  // If no valid point, then check near/far counts
426  default: break;
427  }
428  }
429  return result;
430  }
431  virtual Gesture GetGesture()
432  {
433  Gesture result = kError;
434  if (CheckIfGestureOccurred())
435  {
436  // check if gesture ended
437  if (ReadGestureMode() == 0)
438  {
439  uint8_t level = 0;
440  level = (GetGestureFIFOLevel());
441  ReadGestureFIFO(gfifo_data_, level);
442 
443  // reset processing variables
444  index_ = 0;
445  near_count_ = 0;
446  far_count_ = 0;
447  for (int i = 0; i < 8; i++)
448  {
449  points_[i] = Direction::kNone;
450  }
451 
452  ProcessGestureData(level);
453 
454  result = DecodeGestureData(level);
455  }
456  }
457  return result;
458  }
