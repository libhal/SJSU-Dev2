17  {
18  __asm__ __volatile__(
19  // CPACR is located at address 0xE000ED88
20  "LDR.W R0, =0xE000ED88\n"
21  // Read CPACR
22  "LDR R1, [R0]\n"
23  // Set bits 20-23 to enable CP10 and CP11 coprocessors
24  "ORR R1, R1, #(0xF << 20)\n"
25  // Write back the modified value to the CPACR
26  "STR R1, [R0]\n"
27  // Wait for store to complete
28  "DSB\n"
29  // reset pipeline now the FPU is enabled
30  "ISB\n");
31  }
