91  static void I2cHandler(const Port_t & i2c)
92  {
93  MasterState state = MasterState(i2c.registers->STAT);
94  uint32_t clear_mask = 0;
95  uint32_t set_mask = 0;
96  switch (state)
97  {
98  case MasterState::kBusError: // 0x00
99  {
100  i2c.transaction.status = std::errc::io_error;
101  set_mask = Control::kAssertAcknowledge | Control::kStop;
102  break;
103  }
104  case MasterState::kStartCondition: // 0x08
105  {
106  i2c.registers->DAT = i2c.transaction.GetProperAddress();
107  break;
108  }
109  case MasterState::kRepeatedStart: // 0x10
110  {
111  i2c.transaction.operation = Operation::kRead;
112  i2c.registers->DAT = i2c.transaction.GetProperAddress();
113  break;
114  }
115  case MasterState::kSlaveAddressWriteSentReceivedAck: // 0x18
116  {
117  clear_mask = Control::kStart;
118  if (i2c.transaction.out_length == 0)
119  {
120  i2c.transaction.busy = false;
121  set_mask = Control::kStop;
122  }
123  else
124  {
125  size_t position = i2c.transaction.position++;
126  i2c.registers->DAT = i2c.transaction.data_out[position];
127  }
128  break;
129  }
130  case MasterState::kSlaveAddressWriteSentReceivedNack: // 0x20
131  {
132  clear_mask = Control::kStart;
133  i2c.transaction.busy = false;
134  i2c.transaction.status = std::errc::no_such_device_or_address;
135  set_mask = Control::kStop;
136  break;
137  }
138  case MasterState::kTransmittedDataReceivedAck: // 0x28
139  {
140  if (i2c.transaction.position >= i2c.transaction.out_length)
141  {
142  if (i2c.transaction.repeated)
143  {
144  // OR with 1 to set address as READ for the next transaction
145  i2c.transaction.operation = Operation::kRead;
146  i2c.transaction.position = 0;
147  set_mask = Control::kStart;
148  }
149  else
150  {
151  i2c.transaction.busy = false;
152  set_mask = Control::kStop;
153  }
154  }
155  else
156  {
157  size_t position = i2c.transaction.position++;
158  i2c.registers->DAT = i2c.transaction.data_out[position];
159  }
160  break;
161  }
162  case MasterState::kTransmittedDataReceivedNack: // 0x30
163  {
164  i2c.transaction.busy = false;
165  set_mask = Control::kStop;
166  break;
167  }
168  case MasterState::kArbitrationLost: // 0x38
169  {
170  set_mask = Control::kStart;
171  break;
172  }
173  case MasterState::kSlaveAddressReadSentReceivedAck: // 0x40
174  {
175  clear_mask = Control::kStart;
176  if (i2c.transaction.in_length == 0)
177  {
178  set_mask = Control::kStop;
179  }
180  // If we only want 1 byte, make sure to nack that byte
181  else if (i2c.transaction.in_length == 1)
182  {
183  clear_mask |= Control::kAssertAcknowledge;
184  }
185  // If we want more then 1 byte, make sure to ack the first byte
186  else
187  {
188  set_mask = Control::kAssertAcknowledge;
189  }
190  break;
191  }
192  case MasterState::kSlaveAddressReadSentReceivedNack: // 0x48
193  {
194  clear_mask = Control::kStart;
195  i2c.transaction.status = std::errc::no_such_device_or_address;
196  i2c.transaction.busy = false;
197  set_mask = Control::kStop;
198  break;
199  }
200  case MasterState::kReceivedDataReceivedAck: // 0x50
201  {
202  const size_t kBufferEnd = i2c.transaction.in_length;
203  if (i2c.transaction.position < kBufferEnd)
204  {
205  const size_t kPosition = i2c.transaction.position;
206  i2c.transaction.data_in[kPosition] =
207  static_cast<uint8_t>(i2c.registers->DAT);
208  i2c.transaction.position++;
209  }
210  // Check if the position has been pushed past the buffer length
211  if (i2c.transaction.position + 1 >= kBufferEnd)
212  {
213  clear_mask = Control::kAssertAcknowledge;
214  i2c.transaction.busy = false;
215  }
216  else
217  {
218  set_mask = Control::kAssertAcknowledge;
219  }
220  break;
221  }
222  case MasterState::kReceivedDataReceivedNack: // 0x58
223  {
224  i2c.transaction.busy = false;
225  if (i2c.transaction.in_length != 0)
226  {
227  size_t position = i2c.transaction.position++;
228  i2c.transaction.data_in[position] =
229  static_cast<uint8_t>(i2c.registers->DAT);
230  }
231  set_mask = Control::kStop;
232  break;
233  }
234  case MasterState::kDoNothing: // 0xF8
235  {
236  break;
237  }
238  default:
239  {
240  clear_mask = Control::kStop;
241  SJ2_ASSERT_FATAL(false, "Invalid I2C State Reached!!");
242  break;
243  }
244  }
245  // Clear I2C Interrupt flag
246  clear_mask |= Control::kInterrupt;
247  // Set register controls
248  i2c.registers->CONSET = set_mask;
249  i2c.registers->CONCLR = clear_mask;
250  }

328  void ConfigureClockRate()
329  {
330  // Calculating and setting the I2C Clock rate
331  // Weight the high side duty cycle more than the lower side by 30% in
332  // order to give more time for the bus to charge up.
333  const auto kPeripheralFrequency =
334  sjsu::SystemController::GetPlatformController().GetClockRate(i2c_.id);
335 
336  const float kClockDivider = kPeripheralFrequency / settings.frequency;
337  const float kScll = kClockDivider * settings.duty_cycle;
338  const float kSclh = kClockDivider * (1 - settings.duty_cycle);
339 
340  i2c_.registers->SCLL = static_cast<uint32_t>(kScll);
341  i2c_.registers->SCLH = static_cast<uint32_t>(kSclh);
342  }

347  void BlockUntilFinished() const
348  {
349  if (!IsEnabled())
350  {
351  throw Exception(
352  std::errc::operation_not_permitted,
353  "Attempt to use I2C, before peripheral was not INITIALIZED! Be sure "
354  "to run the i2c.Initialize() method first");
355  }
356 
357  auto wait_for_i2c_transaction = [this]() -> bool {
358  return !i2c_.transaction.busy;
359  };
360 
361  auto wait_status = Wait(i2c_.transaction.timeout, wait_for_i2c_transaction);
362 
363  if (i2c_.transaction.status == CommonErrors::kBusError)
364  {
365  throw CommonErrors::kBusError;
366  }
367  else if (i2c_.transaction.status == CommonErrors::kDeviceNotFound)
368  {
369  throw CommonErrors::kDeviceNotFound;
370  }
371  else if (!wait_status)
372  {
373  // Abort I2C communication if this point is reached!
374  i2c_.registers->CONSET = Control::kAssertAcknowledge | Control::kStop;
375 
376  if (i2c_.transaction.out_length == 0 || i2c_.transaction.in_length == 0)
377  {
378  throw CommonErrors::kTimeout;
379  }
380  }
381 
382  // Ensure that start is cleared before leaving this function, to end the
383  // transaction.
384  i2c_.registers->CONCLR = Control::kStart;
385  }
