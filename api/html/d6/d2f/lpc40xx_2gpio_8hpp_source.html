162  static void InterruptHandler()
163  {
164  int triggered_port;
165  if constexpr (IsPlatform(sjsu::build::Platform::lpc40xx))
166  {
167  triggered_port = *InterruptStatus() >> 2;
168  }
169  else
170  {
171  triggered_port = *InterruptStatus() >> 2;
172  }
173 
174  auto * interrupt = InterruptRegister(triggered_port);
175  int status = *interrupt->rising_status | *interrupt->falling_status;
176  int triggered_pin = __builtin_ctz(status);
177 
178  handlers[triggered_port][triggered_pin]();
179  *interrupt->clear |= (1 << triggered_pin);
180  }

280  void AttachInterrupt(InterruptCallback callback, Edge edge) override
281  {
282  if (!IsInterruptPort())

285  "Only port 0 and port 2 can be used as an interrupt.");
286  }

288  if constexpr (IsPlatform(sjsu::build::Platform::lpc17xx))

294  }
295  else // For LPC40xx and host tests
296  {
297  sjsu::InterruptController::GetPlatformController().Enable({
298  .interrupt_request_number = lpc40xx::GPIO_IRQn,
299  .interrupt_handler = InterruptHandler,

301  }
302 
303  handlers[interrupt_index_][pin_] = callback;
304 
305  auto * interrupt = LocalInterruptRegister();
306  if (edge == Edge::kBoth || edge == Edge::kRising)
307  {
308  *interrupt->rising_enable = bit::Set(*interrupt->rising_enable, pin_);
309  }
310  if (edge == Edge::kBoth || edge == Edge::kFalling)
311  {
312  *interrupt->falling_enable = bit::Set(*interrupt->falling_enable, pin_);
313  }
314  }
