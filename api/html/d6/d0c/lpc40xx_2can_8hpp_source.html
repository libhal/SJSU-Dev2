325  void Send(const Message_t & message) const override
326  {
327  LpcRegisters_t registers = ConvertMessageToRegisters(message);
328 
329  // Wait for one of the buffers to be free so we can transmit a message
330  // through it.
331  bool sent = false;
332  while (!sent)
333  {
334  uint32_t status_register = channel_.registers->SR;
335  // Check if any buffer is available.
336  if (bit::Read(status_register, BufferStatus::kTx1Released))
337  {
338  channel_.registers->TFI1 = registers.frame;
339  channel_.registers->TID1 = registers.id;
340  channel_.registers->TDA1 = registers.data_a;
341  channel_.registers->TDB1 = registers.data_b;
342  channel_.registers->CMR = Value(Commands::kSendTxBuffer1);
343  sent = true;
344  }
345  else if (bit::Read(status_register, BufferStatus::kTx2Released))
346  {
347  channel_.registers->TFI2 = registers.frame;
348  channel_.registers->TID2 = registers.id;
349  channel_.registers->TDA2 = registers.data_a;
350  channel_.registers->TDB2 = registers.data_b;
351  channel_.registers->CMR = Value(Commands::kSendTxBuffer2);
352  sent = true;
353  }
354  else if (bit::Read(status_register, BufferStatus::kTx3Released))
355  {
356  channel_.registers->TFI3 = registers.frame;
357  channel_.registers->TID3 = registers.id;
358  channel_.registers->TDA3 = registers.data_a;

372  Message_t Receive() const override
373  {
374  Message_t message;
375 
376  uint32_t frame = channel_.registers->RFS;
377 
378  // Extract all of the information from the message frame
379  bool is_remote_request = bit::Extract(frame, FrameInfo::kRemoteRequest);
380  uint32_t length = bit::Extract(frame, FrameInfo::kLength);
381  uint32_t format = bit::Extract(frame, FrameInfo::kFormat);
382 
383  message.is_remote_request = is_remote_request;
384  message.length = static_cast<uint8_t>(length);
385  message.format = static_cast<Message_t::Format>(format);
386 
387  // Get the frame ID
388  message.id = channel_.registers->RID;
389 
390  // Pull the bytes from RDA into the payload array
391  message.payload[0] = (channel_.registers->RDA >> (0 * 8)) & 0xFF;
392  message.payload[1] = (channel_.registers->RDA >> (1 * 8)) & 0xFF;
393  message.payload[2] = (channel_.registers->RDA >> (2 * 8)) & 0xFF;
394  message.payload[3] = (channel_.registers->RDA >> (3 * 8)) & 0xFF;
395  // Pull the bytes from RDB into the payload array
396  message.payload[4] = (channel_.registers->RDB >> (0 * 8)) & 0xFF;
397  message.payload[5] = (channel_.registers->RDB >> (1 * 8)) & 0xFF;
398  message.payload[6] = (channel_.registers->RDB >> (2 * 8)) & 0xFF;
399  message.payload[7] = (channel_.registers->RDB >> (3 * 8)) & 0xFF;
400 
401  // Release the RX buffer and allow another buffer to be read.
402  channel_.registers->CMR = Value(Commands::kReleaseRxBuffer);
403 
404  return message;
405  }

407  bool SelfTest(uint32_t id) const override
408  {
409  bool success = false;
410 
411  Message_t test_message;
412  test_message.id = id;
413 
414  // Enable reset mode in order to write to registers
415  SetMode(Mode::kReset, true);
416  // Enable self-test mode
417  SetMode(Mode::kSelfTest, true);
418  // Disable reset mode
419  SetMode(Mode::kReset, false);
420  // Write test message to tx buffer 1
421  LpcRegisters_t registers = ConvertMessageToRegisters(test_message);
422 
423  // Set self-test request and send buffer 1
424  while (true)
425  {
426  uint32_t status_register = channel_.registers->SR;
427  // Check if any buffer is available.
428  if (bit::Read(status_register, BufferStatus::kTx1Released))
429  {
430  channel_.registers->TFI1 = registers.frame;
431  channel_.registers->TID1 = registers.id;
432  channel_.registers->TDA1 = 0;
433  channel_.registers->TDB1 = 0;
434  channel_.registers->CMR = Value(Commands::kSelfReceptionSendTxBuffer1);
435  break;
436  }
437  }
438 
439  // Allow time for RX to fire
440  sjsu::Delay(2ms);
441 
442  auto wait_status = Wait(100ms, [this]() { return HasData(); });
443 
444  if (!IsOk(wait_status))
445  {
446  return false;
447  }
448 
449  // Read the message from the rx buffer and enqueue it into the rx queue.
450  Message_t received_message = Receive();
451 
452  // Check if the received message matches the one we sent
453  if (received_message.id == test_message.id)
454  {
455  success = true;
456  }
457 
458  // Disable self-test mode
459  SetMode(Mode::kReset, true);
460  SetMode(Mode::kSelfTest, false);
461  SetMode(Mode::kReset, false);
462 
463  return success;
464  }

472  void SetBaudRate(units::frequency::hertz_t baud) const override
473  {
474  // According to the BOSCH CAN spec, the nominal bit time is divided into 4
475  // time segments. These segments need to be programmed for the internal
476  // bit timing logic/state machine. Refer to the link below for more
477  // details about the importance of these time segments:
478  // http://www.keil.com/dd/docs/datashts/silabs/boschcan_ug.pdf
479  //
480  // Nominal Bit Time : 1 / 100 000 == 10^-5s
481  //
482  // 0_______________________________10^-5
483  // _/ 1 bit \_
484  // \_______________________________/
485  //
486  // | SYNC | PROP | PHASE1 | PHASE2 | BOSCH
487  // | TSCL | TSEG1 | TSEG2 | LPC
488  // ^
489  // sample point (industry standard: 80%)
490 
491  // Hold the results in RAM rather than altering the register directly
492  // multiple times.
493  uint32_t bus_timing = 0;
494 
495  // Used to compensate for positive and negative edge phase errors. Defines
496  // how much the sample point can be shifted.
497  bus_timing = bit::Insert(bus_timing, 3, BusTiming::kSyncJumpWidth);
498 
499  // These time segments determine the location of the "sample point".
500  bus_timing = bit::Insert(bus_timing, 6, BusTiming::kTimeSegment1);
501  bus_timing = bit::Insert(bus_timing, 1, BusTiming::kTimeSegment2);
502 
503  // The above sampling sifts alters the baud rate by the sum, which is 10,
504  // thus we need to increase the baud rate by that amount.
505  const uint32_t kBaudRateAdjust = 10;
506 
507  // Equation found p.563 of the user manual
508  // tSCL = CANsuppliedCLK * ((prescaler * kBaudRateAdjust) - 1)
509  // Configure the baud rate divider
510  auto & system = sjsu::SystemController::GetPlatformController();
511  auto peripheral_frequency = system.GetClockRate(channel_.id);
512  uint32_t prescaler = (peripheral_frequency / (baud * kBaudRateAdjust)) - 1;
513 
514  bus_timing = bit::Insert(bus_timing, prescaler, BusTiming::kPrescalar);
515 
516  if (baud < kStandardBaudRate)
517  {
518  // The bus is sampled 3 times (recommended for low speeds, 100kHz is
519  // considered HIGH).
520  bus_timing = bit::Insert(bus_timing, 1, BusTiming::kSampling);
521  }
522  else
523  {
524  bus_timing = bit::Insert(bus_timing, 0, BusTiming::kSampling);
525  }
526 
527  channel_.registers->BTR = bus_timing;
528  }

534  LpcRegisters_t ConvertMessageToRegisters(const Message_t & message) const
535  {
536  LpcRegisters_t registers;
537 
538  uint32_t frame_info = 0;
539  frame_info = bit::Insert(frame_info, message.length, FrameInfo::kLength);
540  frame_info = bit::Insert(frame_info, message.is_remote_request,
541  FrameInfo::kRemoteRequest);
542  frame_info =
543  bit::Insert(frame_info, Value(message.format), FrameInfo::kFormat);
544 
545  uint32_t data_a = 0;
546  data_a |= message.payload[0] << (0 * 8);
547  data_a |= message.payload[1] << (1 * 8);
548  data_a |= message.payload[2] << (2 * 8);
549  data_a |= message.payload[3] << (3 * 8);
550 
551  uint32_t data_b = 0;
552  data_b |= message.payload[4] << (0 * 8);
553  data_b |= message.payload[5] << (1 * 8);
554  data_b |= message.payload[6] << (2 * 8);
555  data_b |= message.payload[7] << (3 * 8);
556 
557  registers.frame = frame_info;
558  registers.id = message.id;
559  registers.data_a = data_a;
560  registers.data_b = data_b;
561 
562  return registers;
