228  void Start()
229  {
230  SJ2_ASSERT_FATAL(
231  task_count_ != 0,

233  "are currently scheduled.");

239  private:
244  static void RunTask(void * task_pointer)
245  {
246  TaskInterface & task = *(reinterpret_cast<TaskInterface *>(task_pointer));
247  TaskScheduler & task_scheduler = *(task.GetTaskScheduler());
248 
249  const uint8_t kTaskCount = task_scheduler.GetTaskCount();
250  const uint8_t kTaskIndex = task_scheduler.GetTaskIndex(task.GetName());
251  SJ2_ASSERT_FATAL(kTaskIndex < kTaskCount,
252  "The task index should not exceed the task count.");
253  // Perform PreRun for the task and then set the event group sync bit to
254  // broadcast this task's PreRun has completed
255  EventGroupHandle_t pre_run_event_group_handle =
256  task_scheduler.GetPreRunEventGroupHandle();
257  const EventBits_t kPreRunSyncBits = task_scheduler.GetPreRunSyncBits();
258  const uint32_t kSyncBit = (1 << kTaskIndex);
259  SJ2_ASSERT_FATAL(task.PreRun(),
260  "PreRun() failed for task: %s, terminating scheduler!",
261  task.GetName());
262  // wait for all other PreRun() of other tasks to finish
263  xEventGroupSync(pre_run_event_group_handle, kSyncBit, kPreRunSyncBits,
264  portMAX_DELAY);
265  // All PreRun() complete, each Task's Run() can now start executing...
266  TickType_t last_wake_time = xTaskGetTickCount();
267 
268  while (true)
269  {
270  if (!task.Run())
271  {
272  LogWarning(
273  "An error occurred, the following task will be suspended: %s",
274  task.GetName());
275  vTaskSuspend(NULL);
276  }
277  // delay task if the task's delay time has been set...
278  uint32_t delay_time = task.GetDelayTime();
279  if (delay_time)
280  {
281  vTaskDelayUntil(&last_wake_time, delay_time);
282  }
283  }
284  }

289  void InitializeAllTasks()
290  {
291  for (uint32_t i = 0; i < task_count_; i++)
292  {
293  TaskInterface * task = task_list_[i];
294  *(task->GetHandle()) = xTaskCreateStatic(
295  RunTask, // function to execute the task
296  task->GetName(),
297  static_cast<uint16_t>(StackSize(task->GetStackSize())),
298  PassParameter(task), // pointer of task to run
299  task->GetPriority(),
300  task->GetStack(), // the task's statically allocated memory
301  task->GetTaskBuffer()); // task TCB
302  SJ2_ASSERT_FATAL(task->GetHandle() != nullptr,
303  "Unable to create task: %s", task->GetName());
304  SJ2_ASSERT_FATAL(task->Setup(), "Failed to complete Setup() for task: %s",
305  task->GetName());
306  pre_run_sync_bits_ |= (1 << i);
307  }
308  pre_run_event_group_handle_ =
309  xEventGroupCreateStatic(&pre_run_event_group_buffer_);
310  SJ2_ASSERT_FATAL(pre_run_event_group_handle_ != nullptr,
311  "Failed to create PreRun Event Group!");
312  }

473  sjsu::LogDebug("Start of Heap : 0x%08X", start_of_heap);
474  sjsu::LogDebug("End of Heap : 0x%08X", end_of_heap);
475
