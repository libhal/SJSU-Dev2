103 template <typename T>
104 [[nodiscard]] constexpr T Extract(T target,
105  uint32_t position,
106  uint32_t width = 1)
107 {
108  // Check the types at compile time
109  static_assert(std::numeric_limits<T>::is_integer,
110  "Extract only accepts intergers.");
111  // Need to use an unsigned version of the type T for the mask to make sure
112  // that the shift right doesn't result in a sign extended shift.
113  using UnsignedT = typename std::make_unsigned<T>::type;
114  // At compile time, generate variable containing all 1s with the size of the
115  // target parameter.
116  constexpr UnsignedT kFieldOfOnes = std::numeric_limits<UnsignedT>::max();
117  // At compile time calculate the number of bits in the target parameter.
118  constexpr size_t kTargetWidth = sizeof(T) * 8;
119  // Create mask by shifting the set of 1s down so that the number of 1s from
120  // bit position 0 is equal to the width parameter.
121  UnsignedT mask =
122  static_cast<UnsignedT>(kFieldOfOnes >> (kTargetWidth - width));
123  // Shift target down to the right to get the bit position in the 0th bit
124  // location.
125  // Mask the value to clear any bit after the width's amount of bits.
126  return static_cast<T>((target >> position) & mask);
127 }

203  using UnsignedT = typename std::make_unsigned<T>::type;
204  // At compile time, generate variable containing all 1s with the size of the
205  // target parameter.

210  // bit position 0 is equal to the width parameter.

213  // Clear width's number of bits in the target value at the bit position
214  // specified.

406 constexpr T StreamExtract(const uint8_t * stream,
407  size_t size,
408  Mask mask,
409  sjsu::Endian endian = sjsu::Endian::kLittle)
410 {
411  static_assert(
412  std::is_same<T, uint8_t>::value || std::is_same<T, uint16_t>::value ||
413  std::is_same<T, uint32_t>::value || std::is_same<T, uint64_t>::value,
414  "only unsigned integers are allowed to be used in "
415  "sjsu::bit::StreamExtract().");
416 
417  // Diving the starting position by 8 will give you the starting byte position.
418  // The fractional part is truncated, which gives the correct answer. If it
419  // were rounded, the answer would be incorrect and would require the use of
420  // the floor() function.
421  const size_t kStartByte = mask.position / 8;
422  // We need to figure out if the bit position is not byte aligned. We can do
423  // this by taking the modulus of the mask's bit starting position.
424  const size_t kStartingBitOffset = mask.position % 8;
425  // We need to figure out if the bit position of the last bits not byte
426  // aligned. This is simply to determine if we need to consume an extra byte
427  // from the stream.
428  const size_t kEndingBitOffset = (mask.position + mask.width) % 8;
429 
430  // Contains the total number of bytes that will need to extracted from the
431  // byte stream. The number of bytes can be calculated by taking the bit width
432  // desired and dividing it by the size of a byte (8-bits).
433  size_t total_bytes = mask.width / 8;
434 
435  if (kStartingBitOffset != 0)
436  {
437  total_bytes++;
438  }
439 
440  if (kEndingBitOffset != 0)
441  {
442  total_bytes++;
443  }
444 
445  T result = 0;
446 
447  // A pointer to the starting byte of the sequence to pull bytes from.
448  const uint8_t * next_byte_ptr = &stream[kStartByte];
449  // The direction in which we will pull bytes. This is assumed BIG Endian,
450  // thus, the LSB is the first byte. Going forward will pull out more
451  // significant bytes.
452  int scan_direction = 1;
453 
454  if (endian == sjsu::Endian::kLittle)
455  {
456  // Little Endian has its LSB at the end of the array, thus we will pull
457  // bytes from the end of the byte array.
458  next_byte_ptr = &stream[(size - 1) - kStartByte];
459  // To get the more significate bytes, we need to progress through the array
460  // backwards, so we need to increase backwards.
461  scan_direction = -1;
462  }
463 
464  for (uint32_t i = 0; i < total_bytes && kStartByte + i < size; i++)
465  {
466  const int kShiftAmount = (8 * i) - kStartingBitOffset;
467  const uint8_t kNextByte = *next_byte_ptr;
468  next_byte_ptr += scan_direction;
469  T finished_byte = 0;
470 
471  if (kShiftAmount > 0)
472  {
473  finished_byte = kNextByte << kShiftAmount;
474  }
475  else
476  {
477  finished_byte = kNextByte >> -kShiftAmount;
478  }
479  result |= finished_byte;
480  }
481 
482  Mask trimming_mask = mask;
483  trimming_mask.position = 0;
484  result = Extract(result, trimming_mask);
485 
486  return result;
487 }
