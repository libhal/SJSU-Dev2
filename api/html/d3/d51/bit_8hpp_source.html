50 template <typename T>
51 [[nodiscard]] constexpr T Extract(T target,
52  uint32_t position,
53  uint32_t width = 1)
54 {
55  // Check the types at compile time
56  static_assert(std::numeric_limits<T>::is_integer,
57  "Extract only accepts intergers.");
58  // Need to use an unsigned version of the type T for the mask to make sure
59  // that the shift right doesn't result in a sign extended shift.
60  using UnsignedT = typename std::make_unsigned<T>::type;
61  // At compile time, generate variable containing all 1s with the size of the
62  // target parameter.
63  constexpr UnsignedT kFieldOfOnes = std::numeric_limits<UnsignedT>::max();
64  // At compile time calculate the number of bits in the target parameter.
65  constexpr size_t kTargetWidth = sizeof(T) * 8;
66  // Create mask by shifting the set of 1s down so that the number of 1s from
67  // bit position 0 is equal to the width parameter.
68  UnsignedT mask = kFieldOfOnes >> (kTargetWidth - width);
69  // Shift target down to the right to get the bit position in the 0th bit
70  // location.
71  // Mask the value to clear any bit after the width's amount of bits.
72  return static_cast<T>((target >> position) & mask);
73 }

98  uint32_t position,
99  uint32_t width = 1)
100 {
101  // Check the types at compile time
102  static_assert(
103  std::numeric_limits<T>::is_integer,
104  "1st parameter (target) of Insert function must be an interger.");
105  static_assert(
106  std::numeric_limits<U>::is_integer,
107  "2nd parameter (value) of Insert function must be an interger.");
108  static_assert(sizeof(T) >= sizeof(U),
109  "2nd parameter (value) must be smaller than or equal to the "
110  "size of the target value.");
111  // Need to use an unsigned version of the type T for the mask to make sure
112  // that the shift right doesn't result in a sign extended shift.
113  using UnsignedT = typename std::make_unsigned<T>::type;
114  // At compile time, generate variable containing all 1s with the size of the
115  // target parameter.
116  constexpr UnsignedT kFieldOfOnes = std::numeric_limits<UnsignedT>::max();
117  // At compile time calculate the number of bits in the target parameter.
118  constexpr size_t kTargetWidth = sizeof(T) * 8;
119  // Create mask by shifting the set of 1s down so that the number of 1s from
120  // bit position 0 is equal to the width parameter.
121  UnsignedT mask = kFieldOfOnes >> (kTargetWidth - width);
122  // Clear width's number of bits in the target value at the bit position
123  // specified.
124  target &= ~(mask << position);
125  // AND value with mask to remove any bits beyond the specified width.
126  // Shift masked value into bit position and OR with target value.
127  target |= (value & mask) << position;
128  return static_cast<T>(target);
129 }
