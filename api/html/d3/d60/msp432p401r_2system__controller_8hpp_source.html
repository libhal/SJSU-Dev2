552  Control1Register::kAuxiliaryClockSourceSelect,

557  };
558  uint8_t select_value = Value(source);
559 
560  switch (clock)
561  {
562  case Clock::kMaster: [[fallthrough]];
563  case Clock::kSubsystemMaster: [[fallthrough]];
564  case Clock::kLowSpeedSubsystemMaster: break;
565  case Clock::kAuxiliary:
566  {

572  break;
573  }
574  case Clock::kBackup:
575  {
576  switch (source)
577  {
578  case Oscillator::kLowFrequency: break;
579  case Oscillator::kReference:
580  {
581  select_value = 0b1;
582  break;
583  }
584  default:
585  SJ2_ASSERT_FATAL(false,
586  "The backup clock can only be driven by "
587  "kLowFrequency or kReference.");
588  return;
589  }
590  break;
591  }
592  default:
593  {
594  SJ2_ASSERT_FATAL(
595  false,
596  "clock must be one of the five primary clocks: kAuxiliary, "
597  "kMaster, kSubsystemMaster, kLowSpeedSubsystemMaster, or "
598  "kBackup.");
599  return;
600  }
601  }
602 
603  Control1Register::Register()
604  .Insert(select_value, kPrimaryClockSelectMasks[Value(clock)])

614  units::frequency::hertz_t ConfigureDcoClock() const
615  {
616  const auto & dco_config = clock_configuration_.dco;
617  const units::frequency::kilohertz_t kTargetFrequency = dco_config.frequency;
618 
619  // Configure DCO only if it is used as a clock source.
620  if (dco_config.enabled)
621  {
622  // =======================================================================
623  // Step 1. Ensure the target frequency is valid.
624  // =======================================================================
625  constexpr units::frequency::megahertz_t kMinimumDcoFrequency = 1_MHz;
626  constexpr units::frequency::megahertz_t kMaximumDcoFrequency = 48_MHz;
627  SJ2_ASSERT_FATAL(
628  (kMinimumDcoFrequency <= kTargetFrequency) &&
629  (kTargetFrequency <= kMaximumDcoFrequency),
630  "The desired frequency must be between 1 MHz and 48 MHz.");
631 
632  // =======================================================================
633  // Step 2. Determine the DCO tuning configuration values by finding the
634  // DCO frequency range, DCO constant, and DCO calibration values
635  // based on the desired target frequency.
636  // =======================================================================
637  uint8_t dco_frequency_select = 0b000;
638  float dco_constant =
639  static_cast<float>(device_descriptors->DCOIR_CONSTK_RSEL04);
640  int32_t dco_calibration =
641  static_cast<int32_t>(device_descriptors->DCOIR_FCAL_RSEL04);
642 
643  if (kTargetFrequency >= 32_MHz)
644  {
645  // target frequency is 32 MHz to 48 MHz
646  dco_frequency_select = 0b101;
647  dco_constant =
648  static_cast<float>(device_descriptors->DCOIR_CONSTK_RSEL5);
649  dco_calibration =
650  static_cast<int32_t>(device_descriptors->DCOIR_FCAL_RSEL5);
651  }
652  else if (kTargetFrequency >= 16_MHz)
653  {
654  // target frequency is 16 MHz to 32 MHz
655  dco_frequency_select = 0b100;
656  }
657  else if (kTargetFrequency >= 8_MHz)
658  {
659  // target frequency is 8 MHz to 16 MHz
660  dco_frequency_select = 0b011;
661  }
662  else if (kTargetFrequency >= 4_MHz)
663  {
664  // target frequency is 4 MHz to 8 MHz
665  dco_frequency_select = 0b010;
666  }
667  else if (kTargetFrequency >= 2_MHz)
668  {
669  // target frequency is 2 MHz to 4 MHz
670  dco_frequency_select = 0b001;
671  }
672  else
673  {
674  // target frequency is 1 MHz to 2 MHz with center frequency of 1.5 MHz
675  // use the default initialized values
676  }
677 
678  // =======================================================================
679  // Step 3. Calculate the signed 10-bit tuning value using Equation 6 from
680  // https://www.ti.com/lit/ug/slau356i/slau356i.pdf#page=387
681  // =======================================================================
682  constexpr units::frequency::kilohertz_t kDcoCenterFrequencies[] = {
683  1500_kHz, 3_MHz, 6_MHz, 12_MHz, 24_MHz, 48_MHz
684  };
685  const auto kCenterFrequency = kDcoCenterFrequencies[dco_frequency_select];
686  const float kFrequencyDifference =
687  (kTargetFrequency - kCenterFrequency).to<float>();
688  const float kCalibration =
689  (1.0f + dco_constant * static_cast<float>(768 - dco_calibration));
690 
691  const float kDividend = kFrequencyDifference * kCalibration;
692  const float kDivisor = kTargetFrequency.to<float>() * dco_constant;
693 
694  const int16_t kTuningValue = static_cast<int16_t>(kDividend / kDivisor);
695 
696  // =======================================================================
697  // Step 4. Configure the CSCTL0 register.
698  // =======================================================================
699  UnlockClockSystemRegisters();
700  {
701  Control0Register::Register()
702  .Insert(kTuningValue, Control0Register::kTuningSelect)
703  .Insert(dco_frequency_select, Control0Register::kFrequencySelect)
704  .Set(Control0Register::kEnable)
705  .Save();
706  }
707  LockClockSystemRegisters();
708  }
709 
710  return kTargetFrequency;
711  }
