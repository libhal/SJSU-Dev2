211  void SetClockDivider(Clock clock, uint8_t divider) const
212  {
213  SJ2_ASSERT_FATAL(
214  Value(clock) <= Value(Clock::kLowSpeedSubsystemMaster),
215  "Only the following clocks have a ready status: kAuxiliary, kMaster, "
216  "kSubsystemMaster, or kLowSpeedSubsystemMaster.");
217 
218  uint8_t select_value = 0b000;
219  switch (divider)
220  {
221  case 1: select_value = 0b000; break;
222  case 2: select_value = 0b001; break;
223  case 4: select_value = 0b010; break;
224  case 8: select_value = 0b011; break;
225  case 16: select_value = 0b100; break;
226  case 32: select_value = 0b101; break;
227  case 64: select_value = 0b110; break;
228  case 128: select_value = 0b111; break;
229  default:
230  SJ2_ASSERT_FATAL(false,
231  "Divider must be 1, 2, 4, 8, 16, 32, 64, or 128");
232  return;

235  Control1Register::kAuxiliaryClockDividerSelect,

243  clock_system->CTL1, select_value, kDividerSelectMasks[Value(clock)]);

245  LockClockSystemRegisters();
246  WaitForClockReadyStatus(clock);

320  void SetClockSource(Clock clock, Oscillator source) const
321  {
322  constexpr bit::Mask kPrimaryClockSelectMasks[] = {
323  Control1Register::kAuxiliaryClockSourceSelect,
324  Control1Register::kMasterClockSourceSelect,
325  Control1Register::kSubsystemClockSourceSelect,
326  Control1Register::kSubsystemClockSourceSelect,
327  Control1Register::kBackupClockSourceSelect,
328  };
329  uint8_t select_value = Value(source);
330  switch (clock)
331  {
332  case Clock::kMaster: [[fallthrough]];
333  case Clock::kSubsystemMaster: [[fallthrough]];
334  case Clock::kLowSpeedSubsystemMaster: break;
335  case Clock::kAuxiliary:
336  SJ2_ASSERT_FATAL(
337  Value(source) <= Value(Oscillator::kReference),
338  "The auxiliary clock can only be driven by kLowFrequency, "
339  "kVeryLowFrequency, or kReference.");
340  break;
341  case Clock::kBackup:
342  switch (source)
343  {
344  case Oscillator::kLowFrequency: break;
345  case Oscillator::kReference: select_value = 0b1; break;
346  default:
347  SJ2_ASSERT_FATAL(false,
348  "The backup clock can only be driven by the "
349  "kLowFrequency or kReference.");
350  return;
351  }
352  break;
353  default:
354  SJ2_ASSERT_FATAL(
355  false,
356  "clock must be one of the five primary clocks: kAuxiliary, "
357  "kMaster, kSubsystemMaster, kLowSpeedSubsystemMaster, or kBackup.");
358  return;
359  }
360  clock_system->CTL1 = bit::Insert(clock_system->CTL1,
361  select_value,
362  kPrimaryClockSelectMasks[Value(clock)]);
363  }

373  DcoConfiguration_t CalculateDcoConfiguration(
374  units::frequency::kilohertz_t frequency) const
375  {
376  constexpr units::frequency::megahertz_t kMinimumDcoSpeed = 1_MHz;
377  constexpr units::frequency::megahertz_t kMaximumDcoSpeed = 48_MHz;
378  SJ2_ASSERT_FATAL(
379  (kMinimumDcoSpeed <= frequency) && (frequency <= kMaximumDcoSpeed),
380  "The desired frequency must be between 1 MHz and 48 MHz.");
381  // Determine the DCO tuning configuration values by finding the DCO
382  // frequency range, DCO constant, and DCO calibration values based on the
383  // desired target frequency.
384  constexpr units::frequency::kilohertz_t kDcoCenterFrequencies[] = {
385  1500_kHz, 3_MHz, 6_MHz, 12_MHz, 24_MHz, 48_MHz
386  };
387  uint8_t dco_frequency_select = 0b000;
388  float dco_constant =
389  static_cast<float>(device_descriptors->DCOIR_CONSTK_RSEL04);
390  int32_t dco_calibration =
391  static_cast<int32_t>(device_descriptors->DCOIR_FCAL_RSEL04);
392 
393  if (frequency >= 32_MHz)
394  {
395  // target frequency is 32 MHz to 48 MHz
396  dco_frequency_select = 0b101;
397  dco_constant = static_cast<float>(device_descriptors->DCOIR_CONSTK_RSEL5);
398  dco_calibration =
399  static_cast<int32_t>(device_descriptors->DCOIR_FCAL_RSEL5);
400  }
401  else if (frequency >= 16_MHz)
402  {
403  // target frequency is 16 MHz to 32 MHz
404  dco_frequency_select = 0b100;
405  }
406  else if (frequency >= 8_MHz)
407  {
408  // target frequency is 8 MHz to 16 MHz
409  dco_frequency_select = 0b011;
410  }
411  else if (frequency >= 4_MHz)
412  {
413  // target frequency is 4 MHz to 8 MHz
414  dco_frequency_select = 0b010;
415  }
416  else if (frequency >= 2_MHz)
417  {
418  // target frequency is 2 MHz to 4 MHz
419  dco_frequency_select = 0b001;
420  }
421  else
422  {
423  // target frequency is 1 MHz to 2 MHz with center frequency of 1.5 MHz
424  // use the default initialized values
425  }
426  // Calculate the signed 10-bit tuning value using Equation 6 from
427  // https://www.ti.com/lit/ug/slau356i/slau356i.pdf#page=387
428  const auto kCenterFrequency = kDcoCenterFrequencies[dco_frequency_select];
429  const float kFrequencyDifference =
430  (frequency - kCenterFrequency).to<float>();
431  const float kCalibration =
432  (1.0f + dco_constant * static_cast<float>(768 - dco_calibration));
433  const int16_t kTuneValue =
434  static_cast<int16_t>((kFrequencyDifference * kCalibration) /
435  (frequency.to<float>() * dco_constant));
436 
437  return DcoConfiguration_t{
438  .frequency_select = dco_frequency_select,
439  .tuning_value = kTuneValue,
440  };
441  }
