46  // From Fig. 3 - Photo Current vs. Illuminance of TEMT6000X01 datasheet,
47  // Solve for x using log(y) = m * log(x) + b, where x is the
48  // illuminance in lux and y is the current in micro-amps.
49  //
50  // Plugging in (20, 10) and (100, 50) to solve for m:
51  // m = ∆log(y) / ∆log(x)
52  // = log(y2 / y1) / log(x2 / x1)
53  // = log(50 / 10) / log(100 / 20)
54  // = log(5) / log(5)
55  // m = 1
56  //
57  // Plugging in (100, 50) to solve for b:
58  // log(y) = log(x) + b
59  // log(50) = log(100) + b
60  // b = -log(2)
61  //
62  // Solve for x:
63  // log(y) = log(x) - log(2)
64  // = log(x / 2)
65  // --> y = x / 2
66  // --> x = 2 * y
67  return units::illuminance::lux_t(2 * kCurrent.to<float>());
68  }
