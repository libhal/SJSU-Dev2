34  {
35  // current = voltage / resistance
36  const units::current::microampere_t kCurrent =
37  (adc_.Voltage() / kPullDownResistance);
38  // From Fig. 3 - Photo Current vs. Illuminance of TEMT6000X01 datasheet,
39  // Solve for x using log(y) = m * log(x) + b, where x is the
40  // illuminance in lux and y is the current in micro-amps.
41  //
42  // Plugging in (20, 10) and (100, 50) to solve for m:
43  // m = ∆log(y) / ∆log(x)
44  // = log(y2 / y1) / log(x2 / x1)
45  // = log(50 / 10) / log(100 / 20)
46  // = log(5) / log(5)
47  // m = 1
48  //
49  // Plugging in (100, 50) to solve for b:
50  // log(y) = log(x) + b
51  // log(50) = log(100) + b
52  // b = -log(2)
53  //
54  // Solve for x:
55  // log(y) = log(x) - log(2)
56  // = log(x / 2)
57  // --> y = x / 2
58  // --> x = 2 * y
59  return units::illuminance::lux_t(2.0f * kCurrent.to<float>());
60  }
