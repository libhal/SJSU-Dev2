62  Acceleration_t Read() override
63  {
64  constexpr uint16_t kBytesPerAxis = 2;
65  constexpr uint8_t kNumberOfAxis = 3;
66 
67  Acceleration_t acceleration = {};
68 
69  uint8_t xyz_data[kBytesPerAxis * kNumberOfAxis];
70 
71  i2c_.WriteThenRead(kAddress,
72  { Value(RegisterMap::kXYZStartAddress) },
73  xyz_data,
74  sizeof(xyz_data));
75 
76  // First X-axis Byte (MSB first)
77  // =========================================================================
78  // Bit 7 | Bit 6 | Bit 5 | Bit 4 | Bit 3 | Bit 2 | Bit 1 | Bit 0
79  // XD15 | XD14 | XD13 | XD12 | XD11 | XD10 | XD9 | XD8
80  //
81  // Final X-axis Byte (LSB)
82  // =========================================================================
83  // Bit 7 | Bit 6 | Bit 5 | Bit 4 | Bit 3 | Bit 2 | Bit 1 | Bit 0
84  // XD7 | XD6 | XD5 | XD4 | XD3 | XD2 | XD1 | XD0
85  //
86  // We simply shift and OR the bytes together to get them into a signed int
87  // 16 value.
88 
89  int16_t x = static_cast<int16_t>(xyz_data[0] << 8 | xyz_data[1]);
90  int16_t y = static_cast<int16_t>(xyz_data[2] << 8 | xyz_data[3]);
91  int16_t z = static_cast<int16_t>(xyz_data[4] << 8 | xyz_data[5]);
92 
93  // Convert the 16 bit value into a floating point value m/S^2
94  constexpr int16_t kMax = std::numeric_limits<int16_t>::max();
95  constexpr int16_t kMin = std::numeric_limits<int16_t>::min();
96 
97  float x_ratio = sjsu::Map(x, kMin, kMax, -1.0f, 1.0f);
98  float y_ratio = sjsu::Map(y, kMin, kMax, -1.0f, 1.0f);
99  float z_ratio = sjsu::Map(z, kMin, kMax, -1.0f, 1.0f);
100 
101  acceleration.x = CurrentSettings().gravity * x_ratio;
102  acceleration.y = CurrentSettings().gravity * y_ratio;
103  acceleration.z = CurrentSettings().gravity * z_ratio;
104 
105  return acceleration;
106  }

109  void ConfigureFullScale()
110  {
111  uint32_t gravity_scale = settings.gravity.to<uint32_t>();
112  uint8_t gravity_code;
113 
114  switch (gravity_scale)
115  {
116  case 2: gravity_code = 0x00; break;
117  case 4: gravity_code = 0x01; break;
118  case 8: gravity_code = 0x02; break;
119  case 16: gravity_code = 0x03; break;
120  default:
121  throw Exception(std::errc::invalid_argument,
122  "Invalid gravity scale. Must be 2g, 4g, 8g or 16g.");
123  }
124 
125  // Write in the full scale range; but leave the self test and high pass
126  // filter config untouched
127  constexpr auto kScaleMask = bit::MaskFromRange(3, 4);
128  uint8_t config;
129  i2c_.WriteThenRead(
130  kAddress, { Value(RegisterMap::kDataConfig) }, &config, 1);
131 
132  config = bit::Insert(config, gravity_code, kScaleMask);
133 
134  i2c_.Write(kAddress, { Value(RegisterMap::kDataConfig), config });
135  }
